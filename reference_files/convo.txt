from PIL import Image, ImageDraw

def create_icon(size):
    img = Image.new('RGBA', (size, size), (0, 0, 0, 0))
    draw = ImageDraw.Draw(img)
    
    # Blue circle background
    margin = int(size * 0.05)
    draw.ellipse([margin, margin, size - margin, size - margin], fill='#1976d2')
    
    # White inner circle
    inner_margin = int(size * 0.35)
    draw.ellipse([inner_margin, inner_margin, size - inner_margin, size - inner_margin], fill='white')
    
    # Blue center dot
    center_margin = int(size * 0.42)
    draw.ellipse([center_margin, center_margin, size - center_margin, size - center_margin], fill='#1976d2')
    
    return img

for size in [16, 48, 128]:
